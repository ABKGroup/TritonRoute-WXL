  ${FLEXROUTE_SRC}
)

  PROPERTIES OUTPUT_NAME "flexroute"
)

  ${FLEXROUTE_HOME}/src
)

  def
  lef

  Threads::Threads
  OpenMP::OpenMP_CXX
  Boost::boost

)
